import httpx
from contextlib import asynccontextmanager
from fastapi import FastAPI, Depends, Request

# 1. Manage the client lifecycle globally
@asynccontextmanager
async def lifespan(app: FastAPI):
    # Initialize a single, high-performance connection pool at startup
    limits = httpx.Limits(max_keepalive_connections=5, max_connections=20)
    async_client = httpx.AsyncClient(limits=limits, timeout=10.0)
    
    # Store it inside the FastAPI state container
    app.state.http_client = async_client
    
    yield
    
    # Clean up and close connection sockets gracefully at shutdown
    await async_client.aclose()

app = FastAPI(lifespan=lifespan)

# 2. Dependency injection to pull the client state safely into any route
def get_http_client(request: Request) -> httpx.AsyncClient:
    return request.app.state.http_client

@app.get("/api/v1/external-data")
async def get_external_data(client: httpx.AsyncClient = Depends(get_http_client)):
    # 🚀 Professional, non-blocking asynchronous call utilizing pre-established pools
    response = await client.get("https://api.example.com/data")
    response.raise_for_status() # Fail early if external API drops
    return response.json()
